templates 아래에 hello.html 파일 만들기

```java
package hello.hello_spring.controller;

import org.springframework.stereotype.Controller;
import org.springframework.ui.Model;
import org.springframework.web.bind.annotation.GetMapping;

@Controller // 컨트롤러 어노테이션. 컨트롤러는 이거 달아줘야함.
public class HelloController {
    @GetMapping("hello") // templates의 hello.html과 연결
    public String hello(Model model){ // model이라는 객체 생성.
        model.addAttribute("data", "mark"); // data는 key값, mark는 vaule값
        return "hello"; // 컨트롤러가 return값을 문자열로 하면, viewResolver가 화면을 찾아서 랜더링한다. resources.templates.{문자열}.html을 찾음
    }
}
```
spring-boot-devtools 라이브러리를 추가하면 html파일을 컴파일만 해주면 서버 재시작 없이 view파일 변경 가능. (메뉴build -> recompile)

## 8. MVC와 템플릿 엔진
2. mvc와 템플릿 엔진: 과거 jsp php가 템플릿 엔진. html을 서버에서 프로그래밍해서 동적으로 바꿔서 브라우저에 전달.
MVC: Model, View, Controller

View는 화면을 그리는데 집중.
Controller와 Model은 비즈니스 로직에 집중.

만약 쿼리파라미터를 넣는 경우
```java
package hello.hello_spring.controller;

import org.springframework.stereotype.Controller;
import org.springframework.ui.Model;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RequestParam;

@Controller
public class HelloController {
    @GetMapping("hello")
    public String hello(Model model){
        model.addAttribute("data", "mark");
        return "hello";
    }

    @GetMapping("hello-mvc") // 이건 주소이름 http://localhost8080/hello-mvc
    // 파라미터에 쿼리 파라미터 어노테이션을 넣으면 ?name = 이승희! 이런 방식으로 요청해줘야함
    public String helloMvc(@RequestParam("name") String name, Model model){
        model.addAttribute("name", name);
        return "hello-template";
    }
}

```
